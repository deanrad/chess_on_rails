147   def move_and_record( move )
148     #lift your piece off the square
149     @piece_last_moved_from_coord = move.from_coord
150     
151     #if a capture is notated such as for enpassant, delete at that coordinate
152     # otherwise delete whats moved upon. If nothing deleted, no worries
153     @piece_last_moved = move.capture_coord ? delete(move.capture_coord) : delete(move.from_coord) 
154     
155     #place any captured piece aside
156     @piece_captured = delete(move.to_coord)
157     @piece_captured_from_coord = move.to_coord if @piece_captured
158     
159     #and move your guy there
